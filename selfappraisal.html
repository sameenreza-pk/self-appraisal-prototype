function calculateScores() {

  let weights = document.querySelectorAll(".weight");

  let ratings = document.querySelectorAll(".rating");

  let scores = document.querySelectorAll(".score");


  let totalWeight = 0;

  let goalScore = 0;


  for (let i = 0; i < weights.length; i++) {

    let w = parseFloat(weights[i].value) || 0;

    let r = parseFloat(ratings[i].value) || 0;


    totalWeight += w;

    let s = (w * r) / 100;

    scores[i].innerText = s.toFixed(2);

    goalScore += s;

  }


  document.getElementById("goalTotal").innerText = goalScore.toFixed(2);


  if (totalWeight !== 100) {

    document.getElementById("weightWarning").innerText =

      "Total weight must equal 100%. Current: " + totalWeight + "%";

  } else {

    document.getElementById("weightWarning").innerText = "";

  }


  let compRatings = document.querySelectorAll(".compRating");

  let compTotal = 0;

  let compCount = 0;


  compRatings.forEach(c => {

    if (c.value) {

      compTotal += parseFloat(c.value);

      compCount++;

    }

  });


  let compAvg = compCount ? compTotal / compCount : 0;

  document.getElementById("compAvg").innerText = compAvg.toFixed(2);


  let overall = (goalScore * 0.6) + (compAvg * 0.4);

  document.getElementById("overallScore").innerText = overall.toFixed(2);


  let band = "-";

  if (overall >= 4.5) band = "Outstanding";

  else if (overall >= 3.5) band = "Exceeds Expectations";

  else if (overall >= 2.5) band = "Meets Expectations";

  else band = "Needs Improvement";


  document.getElementById("performanceBand").innerText = band;

}

}
